# 士绅阶层特权定义
# 这些特权仅限中国文化组国家使用

# 乡村建设特权
# 士绅在乡村建设上的优势，伴随文教事业和赋税削减
shishen_rural_development = {
	estate = nobles_estate

	# 解锁条件：仅限中国文化组国家
	potential = {
		culture = { has_culture_group = culture_group:chinese_group }
	}

	# 激活条件（空表示无限制）
	allow = {}

	# 国家修正效果
	country_modifier = {
		# 提升目标满意度
		nobles_estate_target_satisfaction = medium_privilege_target_satisfaction
		# 提升阶层权力
		global_nobles_estate_power = 0.3
		# 降低最高税率（更难征税，反映赋税削减）
		nobles_estate_max_tax = -0.15
		# 提升文化影响力（文教事业）
		cultural_influence = 0.1
		# 提升建筑效率（乡村建设优势）
		building_construction_speed_modifier = 0.15
		# 提升人口转化速度（文教影响）
		global_pop_conversion_speed_modifier = 0.5
		# 提升识字率（文教事业）
		global_nobles_max_literacy = 5
		global_monthly_literacy = 0.01
		# 社会价值观向传统主义移动
		monthly_towards_traditionalist = societal_value_minor_monthly_move
	}
}

# 徭役免除特权
# 士绅免除徭役，减轻赋税等
shishen_corvee_exemption = {
	estate = nobles_estate

	# 解锁条件：仅限中国文化组国家
	potential = {
		culture = { has_culture_group = culture_group:chinese_group }
	}

	allow = {}

	country_modifier = {
		# 提升目标满意度
		nobles_estate_target_satisfaction = large_privilege_target_satisfaction
		# 提升阶层权力
		global_nobles_estate_power = 0.35
		# 大幅降低最高税率（减轻赋税）
		nobles_estate_max_tax = -0.2
		# 降低维护成本（免除徭役，减少公共工程成本）
		levy_maintenance_modifier = -0.1
		# 提升立法效率（士绅支持）
		legislative_efficiency = 0.1
		# 社会价值观向传统主义移动
		monthly_towards_traditionalist = societal_value_minor_monthly_move
	}
}
