csh_1644_unification.1 = { #进京赶考
	type = country_event
	title = csh_1644_unification.1.title
	desc = csh_1644_unification.1.desc
	historical_info = csh_1644_unification.1.historical_info
	immediate = {
		shandong_csh_list_add1 = yes
	}
	option = { #立即进京
		name = csh_1644_unification.1.a
		historical_option = yes
		custom_tooltip = csh_1644_unification.1.tt1
		trigger_event_silently = {
		 	id = csh_1644_unification.2
		 	days = 7
		}
	}
	option = { #暂缓进京
		name = csh_1644_unification.1.b
		custom_tooltip = csh_1644_unification.1.tt2
		show_as_tooltip = {
			every_in_list = {
				list = list_of_shandong_csh
				limit = { owner = c:MNG }
				change_location_owner = c:CSH
				change_control = control_radical_bonus
			}
		}
		if = {
			limit = { exists = c:MNG }
			every_in_list = {
				list = list_of_shandong_csh
				limit = { owner = c:MNG }
				change_location_owner = c:CSH
				change_control = control_mild_bonus
			}
		}
	}
}
