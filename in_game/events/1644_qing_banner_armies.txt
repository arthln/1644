namespace = qng_1644_banner_army

qng_1644_banner_army.1 = {
	type = country_event
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = { tag = QNG }

			set_variable = { name = qng_1644_banner_cavalry_spawned value = 1 }

			location:shenyang ?= {
				if = {
					limit = { owner = root }
					while = {
						count = 20
						create_sub_unit_with_owner = {
							type = a_age_4_reformation_cavalry
							owner = root
							origin = this
							experience = 100
						}
					}
				}
			}

			every_army = {
				limit = {
					is_land_army = yes
					owner = root
				}
				every_sub_unit = {
					add_subunit_morale = {
						value = subunit_morale
						divide = subunit_morale_percentage
						subtract = subunit_morale
					}
				}
			}

			force_recalc_country_active_status = root
		}
	}

	option = {
		name = qng_1644_banner_army.1.a
	}
}
