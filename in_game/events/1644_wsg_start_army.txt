namespace = wsg_1644_army

wsg_1644_army.1 = {
	type = country_event
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = { tag = WSG }

			set_variable = { name = wsg_1644_start_army_spawned value = 1 }

			location:ningyuan_jinzhou ?= {
				if = {
					limit = { owner = root }
					while = {
						count = 10
						create_sub_unit_with_owner = {
							type = a_age_4_reformation_infantry
							owner = root
							origin = this
							experience = 90
						}
					}
					while = {
						count = 5
						create_sub_unit_with_owner = {
							type = a_age_4_reformation_cavalry
							owner = root
							origin = this
							experience = 90
						}
					}
					while = {
						count = 5
						create_sub_unit_with_owner = {
							type = a_age_4_reformation_artillery
							owner = root
							origin = this
							experience = 90
						}
					}
				}
			}



			force_recalc_country_active_status = root
		}
	}

	option = {
		name = wsg_1644_army.1.a
	}
}
